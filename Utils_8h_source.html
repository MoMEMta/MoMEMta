93 // Finds the real solutions to a*x^4 + b*x^3 + c*x^2 + d*x + e = 0

95 // Appends the solutions to the std::vector roots, making no attempt to check whether the vector is empty.

104 // then to try and factorize this quartic into two quadratic equations, which each then gives up to two roots.

106 // then taking the square root of one of these solution (ie there must be a positive solution).

113 // a20*E1^2 + a02*E2^2 + a11*E1*E2 + a10*E1 + a01*E2 + a00 = 0
114 // b20*E1^2 + b02*E2^2 + b11*E1*E2 + b10*E1 + b01*E2 + b00 = 0

213 inline unsigned int createFlagsBitset(char verbosity, bool subregion, bool retainStateFile, unsigned int level, bool smoothing, bool takeOnlyGridFromFile) {
214 
215  unsigned int flags = 0;
216 
217  static unsigned int opt_subregion = 0x04; // bit 2 (=4)
218  static unsigned int opt_smoothing = 0x08; // bit 3 (=8)
219  static unsigned int opt_retainStateFile = 0x10; // bit 4 (=16)
220  static unsigned int opt_takeOnlyGridFromFile = 0x20; // bit 5 (=32)
221 
222  level <<= 8; // bits 8-31
223  flags |= level | verbosity; // verbosity: bits 0-1
224  if(subregion) flags |= opt_subregion;
225  if(!smoothing) flags |= opt_smoothing; // careful true-false inverted
226  if(retainStateFile) flags |= opt_retainStateFile;
227  if(takeOnlyGridFromFile) flags |= opt_takeOnlyGridFromFile;
228 
229  return flags;
230 }
