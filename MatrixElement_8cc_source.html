102  virtual void work() override {
103  static std::vector<LorentzVector> empty_vector;
104 
105  m_weights->clear();
106 
107  uint64_t n_sols = m_invisibles->size();
108 
109  if (!m_invisibles_ids.empty() && !n_sols)
110  return;
111 
112  std::vector<LorentzVector> particles(m_particles_tags.size());
113  for (size_t index = 0; index < m_particles_tags.size(); index++) {
114  particles[index] = m_particles_tags[index].get<LorentzVector>();
115  }
116 
117  // Loop over all solutions if there's, otherwise use only particles
118  if (!m_invisibles.get() || m_invisibles->empty()) {
119  // No invisibles particles. Use only particles for the matrix element
120  // computation
121  internal_work((*m_partons)[0], empty_vector, 1, particles);
122  } else {
123  // Loop over all invisibles solutions
124  for (size_t i = 0; i < m_invisibles->size(); i++) {
125  internal_work((*m_partons)[i], (*m_invisibles)[i], (*m_invisibles_jacobians)[i], particles);

131  virtual void internal_work(const std::vector<LorentzVector>& partons, const std::vector<LorentzVector>& invisibles, double invisibles_jacobian, const std::vector<LorentzVector> particles) {
132  std::vector<std::pair<int, std::vector<double>>> finalStates;
133  std::vector<int64_t> indexing;
134 
135  for (size_t i = 0; i < m_invisibles_ids.size(); i++) {
136  finalStates.push_back(std::make_pair(m_invisibles_ids[i].pdg_id, toVector(invisibles[i])));
137  indexing.push_back(m_invisibles_ids[i].me_index - 1);
138  }
139 
140  for (size_t i = 0; i < m_particles_ids.size(); i++) {
141  finalStates.push_back(std::make_pair(m_particles_ids[i].pdg_id, toVector(particles[i])));
142  indexing.push_back(m_particles_ids[i].me_index - 1);
143  }
144 
145  // Sort the array taking into account the indexing in the configuration
146  std::vector<int64_t> suite(indexing.size());
147  for (size_t i = 0; i < indexing.size(); i++)
148  suite[i] = i;
149 
150  auto permutations = get_permutations(suite, indexing);
151  apply_permutations(finalStates, permutations);
152 
153  std::vector<std::vector<double>> initialState(partons.size());
154  size_t index = 0;
155  for (const auto& parton: partons)
156  initialState[index++] = toVector(parton);
157 
158  auto result = m_ME->compute(initialState, finalStates);
159 
160  double x1 = std::abs(partons[0].Pz() / (sqrt_s / 2.));
161  double x2 = std::abs(partons[1].Pz() / (sqrt_s / 2.));
162 
163  // Compute flux factor 1/(2*x1*x2*s)
164  double phaseSpaceIn = 1. / (2. * x1 * x2 * SQ(sqrt_s));
165 
166  // Compute phase space density for observed particles (not concerned by the change of variable)
167  // dPhi = |P|^2 sin(theta)/(2*E*(2pi)^3)
168  double phaseSpaceOut = 1.;
169  for (const auto& p: particles) {
170  phaseSpaceOut *= SQ(p.P()) * sin(p.Theta()) / (2.0 * p.E() * CB(2. * M_PI));
171  }
172 
173  double weight = phaseSpaceIn * phaseSpaceOut * invisibles_jacobian;
174  for (const auto& jacobian: m_jacobians) {
175  weight *= (*jacobian);
176  }
177 
178  // PDF
179  double final_weight = 0;
180  for (const auto& me: result) {
181  double pdf1 = use_pdf ? m_pdf->xfxQ2(me.first.first, x1, SQ(M_T)) / x1 : 1;
182  double pdf2 = use_pdf ? m_pdf->xfxQ2(me.first.second, x2, SQ(M_T)) / x2 : 1;
183 
184  final_weight += me.second * pdf1 * pdf2;
185  }
186 
187  final_weight *= weight;
188  m_weights->push_back(final_weight);
189  }
