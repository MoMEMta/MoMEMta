51 std::shared_ptr<std::vector<double>> addPhaseSpacePoints(std::shared_ptr<Pool> pool) {
52  pool->current_module("cuba");
53 
54  auto ps_points = pool->put<std::vector<double>>({"cuba", "ps_points"});
55  auto weight = pool->put<double>({"cuba", "ps_weight"});
56 
57  double step = 1. / (N_PS_POINTS - 1);
58 
59  ps_points->resize(N_PS_POINTS);
60  for (size_t i = 0; i < N_PS_POINTS; i++) {
61  double point = i * step;
62  ps_points->operator[](i) = point;
63  }
64 
65  *weight = 1;
66 
67  return ps_points;
68 }
