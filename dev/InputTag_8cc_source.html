24 std::vector<std::string> split(const std::string& s, const std::string& delimiters) {
25 
26  std::vector<std::string> result;
27 
28  size_t current;
29  size_t next = -1;
30  do
31  {
32  next = s.find_first_not_of(delimiters, next + 1);
33  if (next == std::string::npos)
34  break;
35  next -= 1;
36 
37  current = next + 1;
38  next = s.find_first_of(delimiters, current);
39  result.push_back(s.substr(current, next - current));
40  }
41  while (next != std::string::npos);
42 
43  return result;
44 }

56 bool InputTag::isInputTag(const std::string& tag) {
57  if (tag.find("::") == std::string::npos)
58  return false;
59 
60  if (tag.find("/") != std::string::npos) {
61  auto tags = split(tag, "::");
62  auto rtags = split(tags[1], "/");
63  try {
64  int64_t index = std::stoll(rtags[1]) - 1;
65  if (index < 0)
66  return false;
67 
68  return true;
69  } catch (std::invalid_argument e) {
70  return false;
71  }
72  } else {
73  return true;
74  }
75 }
