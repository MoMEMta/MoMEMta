69 template <typename T, typename... Args> std::shared_ptr<T> Pool::put(const InputTag& tag, Args... args) {
70  auto it = m_storage.find(tag);
71  if (it != m_storage.end()) {
72  if (it->second.valid)
73  throw duplicated_tag_error("A module already produced the tag '" + tag.toString() + "'");
74  // A module already requested this block in read-mode. This will only work if the block does not require a non-trivial constructor:
75  if (sizeof...(Args))
76  throw constructor_tag_error("A module already requested the tag '" + tag.toString()
77  + "' which seems to require a constructor call. This is currently not supported.");
78  // Since the memory is allocated, simply consider the block as valid.
79  it->second.valid = true;
80 
81  // If the block is empty, it's a delayed instantiation. Simply flag the block as valid, and allocate memory for it
82  if (it->second.ptr.empty()) {
83  std::shared_ptr<T> ptr(new T(std::forward<Args>(args)...));
84  it->second.ptr = boost::any(ptr);
85  }
86 
87  } else {
88  it = create<T, Args ...>(tag, true, std::forward<Args>(args)...);
89  }
90 
91  // Update current module description
92  Description& description = get_description();
93  description.outputs.push_back(tag.parameter);
94 
95  return boost::any_cast<std::shared_ptr<T>>(it->second.ptr);
96 }
