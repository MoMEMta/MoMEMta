107  virtual Status work() override {
108 
109  solutions->clear();
110 
111  // Equations to solve:
112  //(1) (p1 + p2)^2 = s12 = M1^2 + M2^2 + 2E1E2 + 2p1xp2x + 2p1yp2y + p1zp2z
113  //(2) p1x = - pTx #Coming from pT neutrino = -pT visible = - (p2 + ISR)
114  //(3) p1y = - pTy #Coming from pT neutrino = -pT visible = - (p2 + ISR)
115  //(4) M1 = 0 -> E1^2 = p1x^2 + p1y^2 + p1z^2
116 
117  // Don't spend time on unphysical part of phase-space
118  if (*s12 > SQ(sqrt_s))
119  return Status::NEXT;
120 
121  const LorentzVector& p2 = m_particle_tags[0].get<LorentzVector>();
122 
123  LorentzVector pT;
124  if (pT_is_met) {
125  pT = - m_met_tag.get<LorentzVector>();
126  } else {
127  pT = p2;
128  for (size_t i = 1; i < m_particle_tags.size(); i++) {
129  pT += m_particle_tags[i].get<LorentzVector>();
130  }
131  }
132 
133  const double p22 = p2.M2();
134 
135  // From eq.(1) p1z = -B*E1 + A
136  // From eq.(4) + eq.(1) (1-B^2)* E1^2 + 2AB* E1 - C = 0
137 
138  const double A = (*s12 - p22 + 2*(pT.Px()*p2.Px() + pT.Py()*p2.Py()))/(2*p2.Pz());
139  const double B = p2.E()/p2.Pz();
140  const double C = SQ(pT.Px()) + SQ(pT.Py());
141 
142  // Solve quadratic a*E1^2 + b*E1 + c = 0
143  const double a = 1 - SQ(B);
144  const double b = 2*A*B;
145  const double c = -C;
146 
147  std::vector<double> E1;
148 
149  solveQuadratic(a, b, c, E1, false);
150 
151  if (E1.size() == 0)
152  return Status::NEXT;
153 
154  for(unsigned int i=0; i<E1.size(); i++){
155  const double e1 = E1.at(i);
156 
157  if (e1 < 0.) continue;
158 
159  LorentzVector p1(-pT.Px(), -pT.Py(), A - B*e1, e1);
160 
161  // Check if solutions are physical
162  LorentzVector tot = p1;
163  for (size_t i = 0; i < m_particle_tags.size(); i++) {
164  tot += m_particle_tags[i].get<LorentzVector>();
165  }
166  double q1Pz = std::abs(tot.Pz() + tot.E()) / 2.;
167  double q2Pz = std::abs(tot.Pz() - tot.E()) / 2.;
168  if(q1Pz > sqrt_s/2 || q2Pz > sqrt_s/2)
169  continue;
170 
171  Solution s { {p1}, computeJacobian(p1, p2), true };
172  solutions->push_back(s);
173  }
174 
175  return solutions->size() > 0 ? Status::OK : Status::NEXT;
176  }

183  double computeJacobian(const LorentzVector& p1, const LorentzVector& p2) {
184 
185  const double E1 = p1.E();
186  const double p1z = p1.Pz();
187 
188  const double E2 = p2.E();
189  const double p2z = p2.Pz();
190 
191  // Some extra info in MadWeight Source/MadWeight/blocks/class_b.f Good luck!!
192 
193  double inv_jac = 4.*SQ(M_PI*sqrt_s)*( p2z*E1 - E2*p1z);
194 
195  return 1. / std::abs(inv_jac);
196  }
