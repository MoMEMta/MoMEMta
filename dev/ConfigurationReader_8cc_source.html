32 ConfigurationReader::ConfigurationReader(const std::string& file) {
33 
34  LOG(debug) << "Parsing LUA configuration from " << file;
35  lua_state = lua::init_runtime(this);
36 
37  // Parse file
38  if (luaL_dofile(lua_state.get(), file.c_str())) {
39  std::string error = lua_tostring(lua_state.get(), -1);
40  LOG(fatal) << "Failed to parse configuration file: " << error;
41  throw lua::invalid_configuration_file(error);
42  }
43 
44  // FIXME: Find a better way of doing that
45 
46  // Read global parameters from global variable named 'parameters'
47  configuration.global_parameters.reset(new LazyParameterSet(lua_state, "parameters"));
48  int type = lua_getglobal(lua_state.get(), "parameters");
49  if (type == LUA_TTABLE) {
50  LOG(debug) << "Parsing global parameters.";
51  configuration.global_parameters->parse(lua_state.get(), -1);
52  }
53  lua_pop(lua_state.get(), 1);
54 
55  // Read cuba configuration
56  configuration.cuba_configuration.reset(new LazyParameterSet(lua_state, "cuba"));
57  type = lua_getglobal(lua_state.get(), "cuba");
58  if (type == LUA_TTABLE) {
59  LOG(debug) << "Parsing cuba configuration.";
60  configuration.cuba_configuration->parse(lua_state.get(), -1);
61  }
62  lua_pop(lua_state.get(), 1);
63 
64  for (auto& m: configuration.modules) {
65  LOG(debug) << "Configuration declared module " << m.type << "::" << m.name;
66 
67  lua_getglobal(lua_state.get(), m.type.c_str());
68  lua_getfield(lua_state.get(), -1, m.name.c_str());
69 
70  m.parameters.reset(new ParameterSet(m.type, m.name));
71  m.parameters->parse(lua_state.get(), -1);
72 
73  lua_pop(lua_state.get(), 2);
74  }
75 }
