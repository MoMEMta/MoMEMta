25 bool solveQuadratic(const double a, const double b, const double c, std::vector<double>& roots,
26  bool verbose) {
27 
28  if (!a) {
29  if (!b) {
30  cout << "No solution to equation " << a << " x^2 + " << b << " x + " << c << endl
31  << endl;
32  return false;
33  }
34  roots.push_back(-c / b);
35  if (verbose)
36  cout << "Solution of " << b << " x + " << c << ": " << roots[0]
37  << ", test = " << b * roots[0] + c << endl
38  << endl;
39  return true;
40  }
41 
42  const double rho = SQ(b) - 4. * a * c;
43 
44  if (rho >= 0.) {
45  if (b == 0.) {
46  roots.push_back(sqrt(rho) / (2. * a));
47  roots.push_back(-sqrt(rho) / (2. * a));
48  } else {
49  const double x = -0.5 * (b + sign(b) * sqrt(rho));
50  roots.push_back(x / a);
51  roots.push_back(c / x);
52  }
53  if (verbose) {
54  cout << "Solutions of " << a << " x^2 + " << b << " x + " << c << ":" << endl;
55  for (unsigned short i = 0; i < roots.size(); i++)
56  cout << "x" << i << " = " << roots[i]
57  << ", test = " << a * SQ(roots[i]) + b * roots[i] + c << endl;
58  cout << endl;
59  }
60  return true;
61  } else {
62  if (verbose)
63  cout << "No real solutions to " << a << " x^2 + " << b << " x + " << c << endl << endl;
64  return false;
65  }
66 }

68 bool solveCubic(const double a, const double b, const double c, const double d,
69  std::vector<double>& roots, bool verbose) {
70 
71  if (a == 0)
72  return solveQuadratic(b, c, d, roots, verbose);
73 
74  const double an = b / a;
75  const double bn = c / a;
76  const double cn = d / a;
77 
78  const double Q = SQ(an) / 9. - bn / 3.;
79  const double R = CB(an) / 27. - an * bn / 6. + cn / 2.;
80 
81  if (SQ(R) < CB(Q)) {
82  const double theta = acos(R / sqrt(CB(Q))) / 3.;
83 
84  roots.push_back(-2. * sqrt(Q) * cos(theta) - an / 3.);
85  roots.push_back(-2. * sqrt(Q) * cosXpm2PI3(theta, 1.) - an / 3.);
86  roots.push_back(-2. * sqrt(Q) * cosXpm2PI3(theta, -1.) - an / 3.);
87  } else {
88  const double A = -sign(R) * cbrt(abs(R) + sqrt(SQ(R) - CB(Q)));
89 
90  double B;
91 
92  if (A == 0.)
93  B = 0.;
94  else
95  B = Q / A;
96 
97  const double x = A + B - an / 3.;
98 
99  roots.push_back(x);
100  roots.push_back(x);
101  roots.push_back(x);
102  }
103 
104  if (verbose) {
105  cout << "Solutions of " << a << " x^3 + " << b << " x^2 + " << c << " x + " << d << ":"
106  << endl;
107  for (unsigned short i = 0; i < roots.size(); i++)
108  cout << "x" << i << " = " << roots[i]
109  << ", test = " << a * CB(roots[i]) + b * SQ(roots[i]) + c * roots[i] + d << endl;
110  cout << endl;
111  }
112 
113  return true;
114 }

116 bool solveQuartic(const double a, const double b, const double c, const double d, const double e,
117  std::vector<double>& roots, bool verbose) {
118 
119  if (!a)
120  return solveCubic(b, c, d, e, roots, verbose);
121 
122  if (!b && !c && !d) {
123  roots.push_back(0.);
124  roots.push_back(0.);
125  roots.push_back(0.);
126  roots.push_back(0.);
127  } else {
128  const double an = b / a;
129  const double bn = c / a - (3. / 8.) * SQ(b / a);
130  const double cn = CB(0.5 * b / a) - 0.5 * b * c / SQ(a) + d / a;
131  const double dn =
132  -3. * QU(0.25 * b / a) + e / a - 0.25 * b * d / SQ(a) + c * SQ(b / 4.) / CB(a);
133 
134  std::vector<double> res;
135  solveCubic(1., 2. * bn, SQ(bn) - 4. * dn, -SQ(cn), res, verbose);
136  short pChoice = -1;
137 
138  for (unsigned short i = 0; i < res.size(); ++i) {
139  if (res[i] > 0) {
140  pChoice = i;
141  break;
142  }
143  }
144 
145  if (pChoice < 0) {
146  if (verbose)
147  cout << "No real solution to " << a << " x^4 + " << b << " x^3 + " << c << " x^2 + "
148  << d << " x + " << e << " (no positive root for the resolvent cubic)." << endl
149  << endl;
150  return false;
151  }
152 
153  const double p = sqrt(res[pChoice]);
154  solveQuadratic(p, SQ(p), 0.5 * (p * (bn + res[pChoice]) - cn), roots, verbose);
155  solveQuadratic(p, -SQ(p), 0.5 * (p * (bn + res[pChoice]) + cn), roots, verbose);
156 
157  for (unsigned short i = 0; i < roots.size(); ++i)
158  roots[i] -= an / 4.;
159  }
160 
161  size_t nRoots = roots.size();
162 
163  if (verbose) {
164  if (nRoots) {
165  cout << "Solutions of " << a << " x^4 + " << b << " x^3 + " << c << " x^2 + " << d
166  << " x + " << e << ":" << endl;
167  for (unsigned short i = 0; i < nRoots; i++)
168  cout << "x" << i << " = " << roots[i] << ", test = "
169  << a * QU(roots[i]) + b * CB(roots[i]) + c * SQ(roots[i]) + d * roots[i] + e
170  << endl;
171  cout << endl;
172  } else {
173  cout << "No real solution to " << a << " x^4 + " << b << " x^3 + " << c << " x^2 + "
174  << d << " x + " << e << endl
175  << endl;
176  }
177  }
178 
179  return nRoots > 0;
180 }

182 bool solve2Quads(const double a20, const double a02, const double a11, const double a10,
183  const double a01, const double a00, const double b20, const double b02,
184  const double b11, const double b10, const double b01, const double b00,
185  std::vector<double>& E1, std::vector<double>& E2, bool verbose) {
186 
187  // The procedure used in this function relies on a20 != 0 or b20 != 0
188  if (a20 == 0. && b20 == 0.) {
189 
190  if (a02 != 0. || b02 != 0.) {
191  // Swapping E1 <-> E2 should suffice!
192  return solve2Quads(a02, a20, a11, a01, a10, a00, b02, b20, b11, b01, b10, b00, E2, E1,
193  verbose);
194  } else {
195  return solve2QuadsDeg(a11, a10, a01, a00, b11, b10, b01, b00, E1, E2, verbose);
196  }
197  }
198 
199  const double alpha = b20 * a02 - a20 * b02;
200  const double beta = b20 * a11 - a20 * b11;
201  const double gamma = b20 * a10 - a20 * b10;
202  const double delta = b20 * a01 - a20 * b01;
203  const double omega = b20 * a00 - a20 * b00;
204 
205  const double a = a20 * SQ(alpha) + a02 * SQ(beta) - a11 * alpha * beta;
206  const double b = 2. * a20 * alpha * delta - a11 * (alpha * gamma + delta * beta) -
207  a10 * alpha * beta + 2. * a02 * beta * gamma + a01 * SQ(beta);
208  const double c = a20 * SQ(delta) + 2. * a20 * alpha * omega -
209  a11 * (delta * gamma + omega * beta) - a10 * (alpha * gamma + delta * beta) +
210  a02 * SQ(gamma) + 2. * a01 * beta * gamma + a00 * SQ(beta);
211  const double d = 2. * a20 * delta * omega - a11 * omega * gamma -
212  a10 * (delta * gamma + omega * beta) + a01 * SQ(gamma) +
213  2. * a00 * beta * gamma;
214  const double e = a20 * SQ(omega) - a10 * omega * gamma + a00 * SQ(gamma);
215 
216  solveQuartic(a, b, c, d, e, E2, verbose);
217 
218  for (unsigned short i = 0; i < E2.size(); ++i) {
219 
220  const double e2 = E2[i];
221 
222  if (beta * e2 + gamma != 0.) {
223  // Everything OK
224 
225  const double e1 = -(alpha * SQ(e2) + delta * e2 + omega) / (beta * e2 + gamma);
226  E1.push_back(e1);
227 
228  } else if (alpha * SQ(e2) + delta * e2 + omega == 0.) {
229  // Up to two solutions for e1
230 
231  std::vector<double> e1;
232 
233  if (!solveQuadratic(a20, a11 * e2 + a10, a02 * SQ(e2) + a01 * e2 + a00, e1, verbose)) {
234 
235  if (!solveQuadratic(b20, b11 * e2 + b10, b02 * SQ(e2) + b01 * e2 + b00, e1,
236  verbose)) {
237  cout << "Error in solve2Quads: there should be at least one solution for e1!"
238  << endl;
239  E1.clear();
240  E2.clear();
241  return false;
242  }
243 
244  } else {
245  // We have either a double, or two roots for e1
246  // In this case, e2 must be twice degenerate!
247  // Since in E2 degenerate roots are grouped, E2[i+1] shoud exist and be equal to e2
248  // We then go straight for i+2.
249 
250  if (i < E2.size() - 1) {
251 
252  if (e2 != E2[i + 1]) {
253  cout << "Error in solve2Quads: if there are two solutions for e1, e2 "
254  "should be degenerate!"
255  << endl;
256  E1.clear();
257  E2.clear();
258  return false;
259  }
260 
261  E1.push_back(e1[0]);
262  E1.push_back(e1[1]);
263  ++i;
264  continue;
265 
266  } else {
267  cout << "Error in solve2Quads: if there are two solutions for e1, e2 should be "
268  "degenerate!"
269  << endl;
270  E1.clear();
271  E2.clear();
272  return false;
273  }
274  }
275 
276  } else {
277  // There is no solution given this e2
278  E2.erase(E2.begin() + i);
279  --i;
280  }
281  }
282 
283  return true;
284 }
