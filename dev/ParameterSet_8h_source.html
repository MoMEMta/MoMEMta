244  template<typename T>
245  void set_helper(const std::string& name, const T& value) {
246  static_assert(
247  std::is_same<T, int64_t>::value ||
248  std::is_same<T, double>::value ||
249  std::is_same<T, bool>::value ||
250  std::is_same<T, std::string>::value ||
251  std::is_same<T, InputTag>::value,
252  "Type not supported"
253  );
254 
255  if (frozen) {
256  LOG(fatal) << "You are not allowed to edit a set once frozen.";
257  throw frozen_error("This ParameterSet is frozen");
258  }
259 
260  auto it = m_set.find(name);
261  // If the element does not exist in the set, we create it
262  // Otherwise, we simply update the value
263  if (it == m_set.end()) {
264  create(name, value);
265  } else {
266  setInternal(name, it->second, value);
267  }
268  }
