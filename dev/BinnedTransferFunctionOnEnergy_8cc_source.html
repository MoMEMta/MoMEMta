146 class BinnedTransferFunctionOnEnergy: public BinnedTransferFunctionOnEnergyBase {
147  public:
148 
149  BinnedTransferFunctionOnEnergy(PoolPtr pool, const ParameterSet& parameters): BinnedTransferFunctionOnEnergyBase(pool, parameters) {
150  m_ps_point = get<double>(parameters.get<InputTag>("ps_point"));
151  }
152 
153  virtual Status work() override {
154  const double rec_E = m_reco_input->E();
155  const double range = GetDeltaRange(rec_E);
156  const double gen_E = rec_E - GetDeltaMax(rec_E) + range * (*m_ps_point);
157  const double delta = rec_E - gen_E;
158 
159  // To change the particle's energy without changing its direction and mass:
160  const double gen_pt = std::sqrt(SQ(gen_E) - SQ(m_reco_input->M())) / std::cosh(m_reco_input->Eta());
161  output->SetCoordinates(
162  gen_pt * std::cos(m_reco_input->Phi()),
163  gen_pt * std::sin(m_reco_input->Phi()),
164  gen_pt * std::sinh(m_reco_input->Eta()),
165  gen_E);
166 
167  // The bin number is a ROOT "global bin number" using a 1D representation of the TH2
168  const int bin = m_th2->FindFixBin(std::min(gen_E, m_fallBackEgenMax), delta);
169 
170  // Compute TF*jacobian, where the jacobian includes the transformation of [0,1]->[range_min,range_max] and d|P|/dE
171  *TF_times_jacobian = m_th2->GetBinContent(bin) * range * dP_over_dE(*output);
172 
173  return Status::OK;
174  }
175 
176  private:
177 
178  // Input
179  Value<double> m_ps_point;
180 
181  // Outputs
182  std::shared_ptr<LorentzVector> output = produce<LorentzVector>("output");
183  std::shared_ptr<double> TF_times_jacobian = produce<double>("TF_times_jacobian");
184 
185  // We assume TF=0 for Egen < lower limit of the TH2, and
186  // TF=constant for Egen > upper limit of the TH2
187  // In the former case, the integrated range is adapted (shortened) to the range where TF!=0
188  inline double GetDeltaRange(const double Erec) const {
189  return GetDeltaMax(Erec) - m_deltaMin;
190  }
191 
192  inline double GetDeltaMax(const double Erec) const {
193  return std::min(m_deltaMax, Erec - m_EgenMin);
194  }
195 };

237  BinnedTransferFunctionOnEnergyEvaluator(PoolPtr pool, const ParameterSet& parameters): BinnedTransferFunctionOnEnergyBase(pool, parameters) {
238  m_gen_input = get<LorentzVector>(parameters.get<InputTag>("gen_particle"));
239  }
240 
241  virtual Status work() override {
242  const double rec_E = m_reco_input->E();
243  const double gen_E = m_gen_input->E();
244  double delta = rec_E - gen_E;
245  if (delta <= m_deltaMin || delta >= m_deltaMax) {
246  *TF_value = 0;
247  return Status::OK;
248  }
249 
250  // The bin number is a ROOT "global bin number" using a 1D representation of the TH2
251  const int bin = m_th2->FindFixBin(std::min(gen_E, m_fallBackEgenMax), delta);
252 
253  // Retrieve TF value
254  *TF_value = m_th2->GetBinContent(bin);
255 
256  return Status::OK;
257  }
258 
259  private:
260 
261  // Input
262  Value<LorentzVector> m_gen_input;
263 
264  // Outputs
265  std::shared_ptr<double> TF_value = produce<double>("TF");
266 };
