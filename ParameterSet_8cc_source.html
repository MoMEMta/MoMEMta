31 void ParameterSet::parse(lua_State* L, int index) {
32 
33  TRACE("[parse] >> stack size = {}", lua_gettop(L));
34  size_t absolute_index = lua::get_index(L, index);
35 
36  lua_pushnil(L);
37  while (lua_next(L, absolute_index) != 0) {
38 
39  std::string key = lua_tostring(L, -2);
40 
41  TRACE("[parse] >> key = {}", key);
42 
43  try {
44  boost::any value;
45  bool lazy = false;
46  std::tie(value, lazy) = parseItem(key, L, -1);
47 
48  m_set.emplace(key, Element(value, lazy));
49  } catch(...) {
50  LOG(emerg) << "Exception while trying to parse parameter " << getModuleType() << "." << getModuleName() << "::" << key;
51  lua_pop(L, 1);
52  std::rethrow_exception(std::current_exception());
53  }
54 
55  lua_pop(L, 1);
56  }
57 
58  TRACE("[parse] << stack size = {}", lua_gettop(L));
59 }

79 void ParameterSet::freeze() {
80  if (frozen)
81  return;
82 
83  frozen = true;
84 
85  for (auto& p: m_set) {
86  auto& element = p.second;
87  try {
88  if (element.lazy) {
89  element.lazy = false;
90  if (element.value.type() == typeid(lua::LazyFunction)) {
91  element.value = boost::any_cast<lua::LazyFunction>(element.value)();
92  } else if (element.value.type() == typeid(lua::LazyTableField)) {
93  element.value = boost::any_cast<lua::LazyTableField>(element.value)();
94  }
95  } else {
96  // Recursion
97  if (element.value.type() == typeid(ParameterSet)) {
98  ParameterSet& s = boost::any_cast<ParameterSet&>(element.value);
99  s.freeze();
100  } else if (element.value.type() == typeid(std::vector<ParameterSet>)) {
101  std::vector<ParameterSet>& v = boost::any_cast<std::vector<ParameterSet>&>(element.value);
102  for (auto& c: v)
103  c.freeze();
104  }
105  }
106  } catch(...) {
107  LOG(emerg) << "Exception while trying to parse parameter " << getModuleType() << "." << getModuleName() << "::" << p.first;
108  std::rethrow_exception(std::current_exception());
109  }
110  }
111 }
