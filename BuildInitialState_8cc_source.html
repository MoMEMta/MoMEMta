88  virtual void work() override {
89 
90  output->clear();
91 
92  std::vector<LorentzVector> particles;
93  for (auto& p: input_particles) {
94  particles.push_back(p.get<LorentzVector>());
95  }
96 
97  if (!invisibles.empty()) {
98 
99  size_t n_sols = invisibles.front()->size();
100  for (const auto& i: invisibles) {
101  if (i->size() != n_sols) {
102  throw std::runtime_error("All invisibles must have the same number of solutions.");
103  }
104  }
105 
106  for (size_t sol_index = 0; sol_index < n_sols; sol_index++) {
107 
108  std::vector<LorentzVector> particles_and_invisibles = particles;
109 
110  for (const auto& i: invisibles) {
111 
112  for (const auto& p: (*i)[sol_index])
113  particles_and_invisibles.push_back(p);
114 
115  }
116 
117  do_compute_initials(particles_and_invisibles);
118  }
119 
120  } else {
121  do_compute_initials(particles);
122  }
123  }

128  std::function<void(const std::vector<LorentzVector>&)> compute_initials_trivial =
129  [&](const std::vector<LorentzVector>& particles) {
130  LorentzVector tot;
131  for (const auto& p: particles)
132  tot += p;
133 
134  double q1Pz = (tot.Pz() + tot.E()) / 2.;
135  double q2Pz = (tot.Pz() - tot.E()) / 2.;
136 
137  std::vector<LorentzVector> partons {
138  LorentzVector(0., 0., q1Pz, std::abs(q1Pz)),
139  LorentzVector(0., 0., q2Pz, std::abs(q2Pz))
140  };
141 
142  output->push_back(partons);
143  };

145  std::function<void(const std::vector<LorentzVector>&)> compute_initials_boost =
146  [&](const std::vector<LorentzVector>& particles) {
147  LorentzVector tot;
148  for (const auto& p: particles)
149  tot += p;
150 
151  // Define boost that puts the transverse total momentum vector in its CoM frame
152  LorentzVector transverse_tot = tot;
153  transverse_tot.SetPz(0);
154  ROOT::Math::XYZVector isr_deBoost_vector( transverse_tot.BoostToCM() );
155 
156  // In the "transverse" CoM frame, use total Pz and E to define initial longitudinal quark momenta
157  const ROOT::Math::Boost isr_deBoost( isr_deBoost_vector );
158  const ROOT::Math::PxPyPzEVector new_tot( isr_deBoost * tot );
159 
160  double q1Pz = (new_tot.Pz() + new_tot.E()) / 2.;
161  double q2Pz = (new_tot.Pz() - new_tot.E()) / 2.;
162 
163  std::vector<LorentzVector> partons {
164  LorentzVector(0., 0., q1Pz, std::abs(q1Pz)),
165  LorentzVector(0., 0., q2Pz, std::abs(q2Pz))
166  };
167 
168  // Boost initial parton momenta by the opposite of the transverse boost needed to put the whole system in its CoM
169  const ROOT::Math::Boost isr_boost( -isr_deBoost_vector );
170  partons[0] = isr_boost*partons[0];
171  partons[1] = isr_boost*partons[1];
172 
173  output->push_back(partons);
174  };
