104  virtual void work() override {
105 
106  const double& ps_point = m_ps_point.get<double>();
107  const LorentzVector& reco_particle = m_input.get<LorentzVector>();
108 
109  const double rec_E = reco_particle.E();
110  const double range = GetDeltaRange(rec_E);
111  const double gen_E = rec_E - GetDeltaMax(rec_E) + range*ps_point;
112  const double delta = rec_E - gen_E;
113 
114  // To change the particle's energy without changing its direction and mass:
115  double gen_pt = std::sqrt(SQ(gen_E) - SQ(reco_particle.M())) / std::cosh(reco_particle.Eta());
116  output->SetCoordinates(
117  gen_pt * std::cos(reco_particle.Phi()),
118  gen_pt * std::sin(reco_particle.Phi()),
119  gen_pt * std::sinh(reco_particle.Eta()),
120  gen_E);
121 
122  // The bin number is a ROOT "global bin number" using a 1D representation of the TH2
123  const int bin = m_th2->FindFixBin(std::min(gen_E, m_fallBackEgenMax), delta);
124  // Compute TF*jacobian, where the jacobian includes the transformation of [0,1]->[range_min,range_max] and dE/d|P|
125  *TF_times_jacobian = m_th2->GetBinContent(bin) * range * dE_over_dP(*output);
126  }
