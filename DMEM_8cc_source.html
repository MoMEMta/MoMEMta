57  virtual void work() override {
58 
59  LorentzVector tot;
60  for(const auto &v: m_particle_tags)
61  tot += v.get<LorentzVector>();
62 
63  // No invisibles -> loop over particles we got as input
64  if(!m_invisibles.get() || m_invisibles->empty()) {
65 
66  m_hist->Fill(tot.M(), (*integrands)[0] * (*psWeight));
67 
68  } else {
69 
70  // Several solutions for the invisibles -> loop over all solution,
71  // fill histogram with corresponding integrand.
72  for(size_t sol = 0; sol < m_invisibles->size(); sol++) {
73  LorentzVector thisTot = tot;
74 
75  for(const auto &v: (*m_invisibles)[sol])
76  thisTot += v;
77 
78  m_hist->Fill(thisTot.M(), (*integrands)[sol] * (*psWeight));
79  }
80 
81  }
82  }
