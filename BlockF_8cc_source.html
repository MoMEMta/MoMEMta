89  virtual void work() override {
90 
91  invisibles->clear();
92  jacobians->clear();
93 
94  // Don't spend time on unphysical part of phase-space
95  if(*s13 > SQ(sqrt_s) || *s24 > SQ(sqrt_s))
96  return;
97 
98  const LorentzVector& p3 = m_particle_tags[0].get<LorentzVector>();
99  const LorentzVector& p4 = m_particle_tags[1].get<LorentzVector>();
100 
101  //leave the variables E2 and p2y as free parameters
102  std::vector<double> E2;
103  std::vector<double> p2y;
104 
105  double p3x = p3.Px();
106  double p3y = p3.Py();
107  double p3z = p3.Pz();
108  double E3 = p3.E();
109  double p4x = p4.Px();
110  double p4y = p4.Py();
111  double p4z = p4.pz();
112  double E4 = p4.E();
113  double p33 = p3.M2();
114  double p44 = p4.M2();
115 
116  double Eb = E3+E4;
117  double pbx = p3x+p4x;
118  double pby = p3y+p4y;
119  double pbz = p3z+p4z;
120 
121  double q1 = m_ps_point1.get<double>();
122  double q2 = m_ps_point2.get<double>();
123 
124  const double Qm = sqrt_s*(q1-q2)/2.;
125  const double Qp = sqrt_s*(q1+q2)/2.;
126 
127  //p1x = alpha1*p2y + beta1*E2 + gamma1
128  //p1y = alpha2*p2y + beta2*E2 + gamma2
129  //p1z = alpha3*p2y + beta3*E2 + gamma3
130  //p2x = alpha4*p2y + beta4*E2 + gamma4
131  //p2z = alpha5*p2y + beta5*E2 + gamma5
132  //E1 = alpha6*p2y + beta6*E2 + gamma6
133 
134  double alpha1 = (-p3z*p4y+p3y*p4z)/(-p3z*p4x+p3x*p4z);
135  double beta1 = (E4*p3z-E3*p4z)/(p3z*p4x+p3x*p4z);
136  double gamma1 = (p44*p3z-2*E3*Eb*p4z+p33*p4z+2*p3z*p4x*pbx+
137  2*p3y*p4z*pby+2*p3z*p4z*pbz-2*p3z*p4z*Qm+2*E3*p4z*Qp-
138  p4z*(*s13)-p3z*(*s24))/(2*(-p3z*p4x+p3x*p4z));
139 
140  double alpha2 = -1;
141  double beta2 = 0;
142  double gamma2 = -pby;
143 
144  double alpha3 = (p3y*p4x-p3x*p4y)/(p3z*p4x-p3x*p4z);
145  double beta3 = (E4*p3x-E3*p4x)/(p3z*p4x-p3x*p4z);
146  double gamma3 = 0.5*((p44*p3x-2*E3*Eb*p4x+p33*p4x+2*p3x*p4x*pbx+
147  2*p3y*p4x*pby+2*p3x*p4z*pbz-2*p3x*p4z*Qm+2*E3*p4x*Qp-
148  p4x*(*s13)-p3x*(*s24))/(p3z*p4x-p3x*p4z));
149 
150  double alpha4 = (p3z*p4y-p3y*p4z)/(-p3z*p4x+p3x*p4z);
151  double beta4 = (-E4*p3z+E3*p4z)/(-p3z*p4x+p3x*p4z);
152  double gamma4 = (-2*p44*p3z+4*E3*Eb*p4z-2*p33*p4z-
153  4*p3z*p4x*pbx-4*p3y*p4z*pby-4*p3z*p4z*pbz+
154  4*p3z*p4z*Qm-4*E3*p4z*Qp+2*p4z*(*s13)+
155  2*p3z*(*s24))/(-4*p3z*p4x+4*p3x*p4z)-pbx;
156 
157  double alpha5 = (-p3y*p4x+p3x*p4y)/(p3z*p4x-p3x*p4z);
158  double beta5 = (-E4*p3x+E3*p4x)/(p3z*p4x-p3x*p4z);
159  double gamma5 = 0.5*((-p44*p3x+2*E3*Eb*p4x-p33*p4x-
160  2*p3x*p4x*pbx-2*p3y*p4x*pby-2*p3z*p4x*pbz+
161  2*p3z*p4x*Qm-2*E3*p4x*Qp+p4x*(*s13)+
162  p3x*(*s24))/(p3z*p4x-p3x*p4z));
163 
164  double alpha6 = 0;
165  double beta6 = -1;
166  double gamma6 = -Eb+Qp;
167 
168  //Put these variables in p1^2=0, p2^2=0. You will get:
169  //a11*p2y^2 + a22*E2^2 + a12*p2y*E2 + a10*p2y + a01*E2 + a00 = 0;
170  //same with bij
171 
172  double a11 = SQ(alpha6) - SQ(alpha1) - SQ(alpha2) - SQ(alpha3);
173  double a22 = SQ(beta6) - SQ(beta1) - SQ(beta2) - SQ(beta3);
174  double a12 = 2*(alpha6*beta6 - alpha1*beta1 - alpha2*beta2 - alpha3*beta3);
175  double a10 = 2*(alpha6*gamma6 - alpha1*gamma1 - alpha2*gamma2 - alpha3*gamma3);
176  double a01 = 2*(beta6*gamma6 - beta1*gamma1 - beta2*gamma2 - beta3*gamma3);
177  double a00 = SQ(gamma6) - SQ(gamma1) - SQ(gamma2) - SQ(gamma3);
178 
179  double b11 = 1 + SQ(alpha4) + SQ(alpha5);
180  double b22 = SQ(beta4) + SQ(beta5) -1;
181  double b12 = 2*(alpha4*beta4 + alpha5*beta5);
182  double b10 = 2*(alpha4*gamma4 + alpha5*gamma5);
183  double b01 = 2*(beta4*gamma4 + beta5*gamma5);
184  double b00 = SQ(gamma4) + SQ(gamma5);
185 
186  solve2Quads(a11, a22, a12, a10, a01, a00, b11, b22, b12, b10, b01, b00, p2y, E2, false);
187  //In this case there will be up to 2 solutions
188 
189  if (p2y.size() == 0)
190  return;
191 
192  for (size_t i=0; i<p2y.size(); i++) {
193  const double e1 = p2y.at(i);
194  const double e2 = E2.at(i);
195 
196  if (e1 < 0. || e2 < 0.)
197  continue;
198 
199  LorentzVector p1(alpha6*e1+beta6*e2+gamma6, //E1
200  alpha1*e1+beta1*e2+gamma1, //p1x
201  alpha2*e1+beta2*e2+gamma2, //p1y
202  alpha3*e1+beta3*e2+gamma3 //p1z
203  );
204 
205  LorentzVector p2(e2, //E2
206  alpha4*e1+beta4*e2+gamma4, //p2x
207  e1, //p2y
208  alpha5*e1+beta5*e2+gamma5 //p2z
209  );
210 
211  // Check if solutions are physical
212  LorentzVector tot = p1 + p2 + p3 + p4;
213  double q1Pz = std::abs(tot.Pz() + tot.E()) / 2.;
214  double q2Pz = std::abs(tot.Pz() - tot.E()) / 2.;
215  if(q1Pz > sqrt_s/2 || q2Pz > sqrt_s/2)
216  continue;
217 
218  invisibles->push_back({p1, p2});
219  jacobians->push_back(computeJacobian(p1, p2, p3, p4));
220  }
221  }

227  double computeJacobian(const LorentzVector& p1, const LorentzVector& p2, const LorentzVector& p3, const LorentzVector& p4) {
228 
229  const double E1 = p1.E();
230  const double p1x = p1.Px();
231  const double p1y = p1.Py();
232  const double p1z = p1.Pz();
233 
234  const double E2 = p2.E();
235  const double p2x = p2.Px();
236  const double p2y = p2.Py();
237  const double p2z = p2.Pz();
238 
239  const double E3 = p3.E();
240  const double p3x = p3.Px();
241  const double p3y = p3.Py();
242  const double p3z = p3.Pz();
243 
244  const double E4 = p4.E();
245  const double p4x = p4.Px();
246  const double p4y = p4.Py();
247  const double p4z = p4.Pz();
248 
249  double inv_jac= (E4*(p1z*p2y*p3x - p1y*p2z*p3x - p1z*p2x*p3y + p1x*p2z*p3y + p1y*p2x*p3z - p1x*p2y*p3z) + E2*p1z*p3y*p4x - E1*p2z*p3y*p4x - E2*p1y*p3z*p4x + E1*p2y*p3z*p4x - E2*p1z*p3x*p4y + E1*p2z*p3x*p4y + E2*p1x*p3z*p4y - E1*p2x*p3z*p4y + (E2*p1y*p3x - E1*p2y*p3x - E2*p1x*p3y + E1*p2x*p3y)*p4z + E3*(-(p1z*p2y*p4x) + p1y*p2z*p4x + p1z*p2x*p4y - p1x*p2z*p4y - p1y*p2x*p4z + p1x*p2y*p4z));
250 
251  return 1./( std::abs(inv_jac) * 4.*16.*pow(TMath::Pi(),2.) );
252  }
