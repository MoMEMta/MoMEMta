177  Type lua_array_unique_type(lua_State* L, int index) {
178 
179  size_t absolute_index = get_index(L, index);
180 
181  if (lua_type(L, absolute_index) != LUA_TTABLE)
182  return NOT_SUPPORTED;
183 
184  Type result = NOT_SUPPORTED;
185 
186  lua_pushnil(L);
187  while (lua_next(L, absolute_index) != 0) {
188  if (result == NOT_SUPPORTED) {
189  result = type(L, -1);
190  } else {
191  Type entry_type = type(L, -1);
192 
193  if ((result == INTEGER) && (entry_type == REAL))
194  result = REAL;
195  else if ((result == REAL) && (entry_type == INTEGER))
196  result = REAL;
197  else if (result != entry_type) {
198  lua_pop(L, 2);
199  result = NOT_SUPPORTED;
200  break;
201  }
202  }
203 
204  lua_pop(L, 1);
205  }
206 
207  return result;

210  std::pair<boost::any, bool> to_any(lua_State* L, int index) {
211 
212  TRACE("[to_any] >> stack size = {}", lua_gettop(L));
213  size_t absolute_index = get_index(L, index);
214 
215  boost::any result;
216  bool lazy = false;
217 
218  auto type = lua_type(L, absolute_index);
219  switch (type) {
220  case LUA_TNUMBER: {
221  if (lua_isinteger(L, absolute_index)) {
222  int64_t number = lua_tointeger(L, absolute_index);
223  result = number;
224  } else {
225  double number = lua_tonumber(L, absolute_index);
226  result = number;
227  }
228  } break;
229 
230  case LUA_TBOOLEAN: {
231  bool value = lua_toboolean(L, absolute_index);
232  result = value;
233  } break;
234 
235  case LUA_TSTRING: {
236  std::string value = lua_tostring(L, absolute_index);
237  if (InputTag::isInputTag(value)) {
238  InputTag tag = InputTag::fromString(value);
239  result = tag;
240  } else {
241  result = value;
242  }
243  } break;
244 
245  case LUA_TTABLE: {
246  TRACE("[to_any::table] >> stack size = {}", lua_gettop(L));
247  if (lua::lua_is_array(L, absolute_index) > 0) {
248 
249  Type type = NOT_SUPPORTED;
250 
251  if ((type = lua::lua_array_unique_type(L, absolute_index)) != NOT_SUPPORTED) {
252  result = to_vector(L, absolute_index, type);
253  } else {
254  throw invalid_array_error("Various types stored into the array. This is not supported.");
255  }
256 
257  } else {
258  ConfigurationSet cfg;
259  cfg.parse(L, absolute_index);
260  result = cfg;
261  }
262  TRACE("[to_any::table] << stack size = {}", lua_gettop(L));
263  } break;
264 
265  case LUA_TFUNCTION: {
266  TRACE("[to_any::function] >> stack size = {}", lua_gettop(L));
267 
268  result = LazyFunction(L, absolute_index);
269  lazy = true;
270 
271  TRACE("[to_any::function] << stack size = {}", lua_gettop(L));
272  } break;
273 
274  default: {
275  LOGGER->critical("Unsupported lua type: {}", lua_typename(L, type));
276  throw lua::invalid_configuration_file("");
277  } break;
278  }
279 
280  TRACE("[to_any] << final type = {}", demangle(result.type().name()));
281  TRACE("[to_any] << stack size = {}", lua_gettop(L));
282  return {result, lazy};
