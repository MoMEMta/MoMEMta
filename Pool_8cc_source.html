38 boost::any Pool::reserve(const InputTag& tag) {
39  auto it = m_storage.find(tag);
40  if (it == m_storage.end()) {
41  // This tag do not exist currently in the pool
42  // Reserve a slot, but mark it as invalid.
43  // Once a module inform the pool it produces such a tag, the slot will
44  // be flagged as valid.
45  PoolContent content { boost::any(), false };
46  it = m_storage.emplace(tag, content).first;
47  }
48 
49  // Update current module description
50  assert(! m_current_module.empty());
51  Description& description = m_description[m_current_module];
52  description.inputs.push_back(tag);
53 
54  return it->second.ptr;
55 }

65 void Pool::alias(const InputTag& from, const InputTag& to) {
66  if (from.isIndexed() || to.isIndexed()) {
67  throw std::invalid_argument("Indexed input tag cannot be passed as argument of the pool. Use the `get` function of the input tag to retrieve its content.");
68  }
69 
70  auto from_it = m_storage.find(from);
71  if (from_it == m_storage.end())
72  throw tag_not_found_error("No such tag in pool: " + from.toString());
73 
74  auto to_it = m_storage.find(to);
75  if (to_it != m_storage.end())
76  throw duplicated_tag_error("A module already produced the tag '" + to.toString() + "'");
77 
78  m_storage[to] = m_storage[from];
79 }
