44  virtual void work() override {

46  invisibles->clear();
47  jacobians->clear();

50  if(*s13 >= *s134 || *s25 >= *s256 || *s13 >= SQ(sqrt_s) || *s134 >= SQ(sqrt_s) || *s25 >= SQ(sqrt_s) || *s256 >= SQ(sqrt_s))

53  const LorentzVector& p3 = m_particle_tags[0].get<LorentzVector>();
54  const LorentzVector& p4 = m_particle_tags[1].get<LorentzVector>();
55  const LorentzVector& p5 = m_particle_tags[2].get<LorentzVector>();
56  const LorentzVector& p6 = m_particle_tags[3].get<LorentzVector>();
57 
58  // FIXME
59  const LorentzVector ISR; // = (*particles)[0];
60 
61  // pT = transverse total momentum of the visible particles
62  // It will be used to reconstruct neutrinos, but we want to take into account the measured ISR (pt_isr = - pt_met - pt_vis),
63  // so we add pt_isr to pt_vis in order to have pt_vis + pt_nu + pt_isr = 0 as it should be.
64  auto pT = p3 + p4 + p5 + p6 + ISR;
65 
66  const double p34 = p3.Dot(p4);
67  const double p56 = p5.Dot(p6);
68  const double p33 = p3.M2();
69  const double p44 = p4.M2();
70  const double p55 = p5.M2();
71  const double p66 = p6.M2();
72 
73  // A1 p1x + B1 p1y + C1 = 0, with C1(E1,E2)
74  // A2 p1y + B2 p2y + C2 = 0, with C2(E1,E2)
75  // ==> express p1x and p1y as functions of E1, E2
76 
77  const double A1 = 2.*( -p3.Px() + p3.Pz()*p4.Px()/p4.Pz() );
78  const double A2 = 2.*( p5.Px() - p5.Pz()*p6.Px()/p6.Pz() );
79 
80  const double B1 = 2.*( -p3.Py() + p3.Pz()*p4.Py()/p4.Pz() );
81  const double B2 = 2.*( p5.Py() - p5.Pz()*p6.Py()/p6.Pz() );
82 
83  const double Dx = B2*A1 - B1*A2;
84  const double Dy = A2*B1 - A1*B2;
85 
86  const double X = 2*( pT.Px()*p5.Px() + pT.Py()*p5.Py() - p5.Pz()/p6.Pz()*( 0.5*(*s25 - *s256 + p66) + p56 + pT.Px()*p6.Px() + pT.Py()*p6.Py() ) ) + p55 - *s25;
87  const double Y = p3.Pz()/p4.Pz()*( *s13 - *s134 + 2*p34 + p44 ) - p33 + *s13;
88 
89  // p1x = alpha1 E1 + beta1 E2 + gamma1
90  // p1y = ...(2)
91  // p1z = ...(3)
92  // p2z = ...(4)
93  // p2x = ...(5)
94  // p2y = ...(6)
95 
96  const double alpha1 = -2*B2*(p3.E() - p4.E()*p3.Pz()/p4.Pz())/Dx;
97  const double beta1 = 2*B1*(p5.E() - p6.E()*p5.Pz()/p6.Pz())/Dx;
98  const double gamma1 = B1*X/Dx + B2*Y/Dx;
99 
100  const double alpha2 = -2*A2*(p3.E() - p4.E()*p3.Pz()/p4.Pz())/Dy;
101  const double beta2 = 2*A1*(p5.E() - p6.E()*p5.Pz()/p6.Pz())/Dy;
102  const double gamma2 = A1*X/Dy + A2*Y/Dy;
103 
104  const double alpha3 = (p4.E() - alpha1*p4.Px() - alpha2*p4.Py())/p4.Pz();
105  const double beta3 = -(beta1*p4.Px() + beta2*p4.Py())/p4.Pz();
106  const double gamma3 = ( 0.5*(*s13 - *s134 + p44) + p34 - gamma1*p4.Px() - gamma2*p4.Py() )/p4.Pz();
107 
108  const double alpha4 = (alpha1*p6.Px() + alpha2*p6.Py())/p6.Pz();
109  const double beta4 = (p6.E() + beta1*p6.Px() + beta2*p6.Py())/p6.Pz();
110  const double gamma4 = ( 0.5*(*s25 - *s256 + p66) + p56 + (gamma1 + pT.Px())*p6.Px() + (gamma2 + pT.Py())*p6.Py() )/p6.Pz();
111 
112  const double alpha5 = -alpha1;
113  const double beta5 = -beta1;
114  const double gamma5 = -pT.Px() - gamma1;
115 
116  const double alpha6 = -alpha2;
117  const double beta6 = -beta2;
118  const double gamma6 = -pT.Py() - gamma2;
119 
120  // a11 E1^2 + a22 E2^2 + a12 E1E2 + a10 E1 + a01 E2 + a00 = 0
121  // id. with bij
122 
123  const double a11 = -1 + ( SQ(alpha1) + SQ(alpha2) + SQ(alpha3) );
124  const double a22 = SQ(beta1) + SQ(beta2) + SQ(beta3);
125  const double a12 = 2.*( alpha1*beta1 + alpha2*beta2 + alpha3*beta3 );
126  const double a10 = 2.*( alpha1*gamma1 + alpha2*gamma2 + alpha3*gamma3 );
127  const double a01 = 2.*( beta1*gamma1 + beta2*gamma2 + beta3*gamma3 );
128  const double a00 = SQ(gamma1) + SQ(gamma2) + SQ(gamma3);
129 
130  const double b11 = SQ(alpha5) + SQ(alpha6) + SQ(alpha4);
131  const double b22 = -1 + ( SQ(beta5) + SQ(beta6) + SQ(beta4) );
132  const double b12 = 2.*( alpha5*beta5 + alpha6*beta6 + alpha4*beta4 );
133  const double b10 = 2.*( alpha5*gamma5 + alpha6*gamma6 + alpha4*gamma4 );
134  const double b01 = 2.*( beta5*gamma5 + beta6*gamma6 + beta4*gamma4 );
135  const double b00 = SQ(gamma5) + SQ(gamma6) + SQ(gamma4);
136 
137  // Find the intersection of the 2 conics (at most 4 real solutions for (E1,E2))
138  std::vector<double> E1, E2;
139  solve2Quads(a11, a22, a12, a10, a01, a00, b11, b22, b12, b10, b01, b00, E1, E2, false);
140 
141  // For each solution (E1,E2), find the neutrino 4-momenta p1,p2
142 
143  if (E1.size() == 0)
144  return;
145 
146  for(unsigned int i=0; i<E1.size(); i++){
147  const double e1 = E1.at(i);
148  const double e2 = E2.at(i);
149 
150  if (e1 < 0. || e2 < 0.)
151  continue;
152 
153  LorentzVector p1(
154  alpha1*e1 + beta1*e2 + gamma1,
155  alpha2*e1 + beta2*e2 + gamma2,
156  alpha3*e1 + beta3*e2 + gamma3,
157  e1);
158 
159  LorentzVector p2(
160  alpha5*e1 + beta5*e2 + gamma5,
161  alpha6*e1 + beta6*e2 + gamma6,
162  alpha4*e1 + beta4*e2 + gamma4,
163  e2);
164 
165  // Check if solutions are physical
166  LorentzVector tot = p1 + p2 + p3 + p4 + p5 + p6;
167  double q1Pz = std::abs(tot.Pz() + tot.E()) / 2.;
168  double q2Pz = std::abs(tot.Pz() - tot.E()) / 2.;
169  if(q1Pz > sqrt_s/2 || q2Pz > sqrt_s/2)
170  continue;
171 
172  invisibles->push_back({p1, p2});
173  jacobians->push_back(computeJacobian(p1, p2, p3, p4, p5, p6));
174  }

227  p1x*p2y*p56z) +

229  p2y*p34x*p56z + p2x*p34y*p56z)) +

235  E2*(p1z*p34y*p56x - p1y*p34z*p56x - p1z*p34x*p56y +

237  E34*(E5*p2z*(p1z*p3y*p56x - p1y*p3z*p56x - p1z*p3x*p56y +
238  p1x*p3z*p56y) +
239  E5*(p1z*p2y*p3x - p1z*p2x*p3y + p1y*p2x*p3z -
240  p1x*p2y*p3z)*p56z -
241  (E56*p2z - E2*p56z)*
242  (p1z*p3y*p5x - p1y*p3z*p5x - p1z*p3x*p5y + p1x*p3z*p5y)
243  - (E56*(p1z*p2y*p3x - p1z*p2x*p3y + p1y*p2x*p3z -
244  p1x*p2y*p3z) +
245  E2*(p1z*p3y*p56x - p1y*p3z*p56x - p1z*p3x*p56y +
246  p1x*p3z*p56y))*p5z) +
247  E1*(E5*(p2z*(-(p34z*p3y*p56x) + p34y*p3z*p56x +
248  p34z*p3x*p56y - p34x*p3z*p56y) +
249  (-(p2y*p34z*p3x) + p2x*p34z*p3y + p2y*p34x*p3z -
250  p2x*p34y*p3z)*p56z) +
251  (E56*p2z - E2*p56z)*
252  (p34z*p3y*p5x - p34y*p3z*p5x - p34z*p3x*p5y +
253  p34x*p3z*p5y) +
