30 ConfigurationReader::ConfigurationReader(const std::string& file) {
31 
32  LOG(debug) << "Parsing LUA configuration from " << file;
33  lua_state = lua::init_runtime(this);
34 
35  // Parse file
36  if (luaL_dofile(lua_state.get(), file.c_str())) {
37  std::string error = lua_tostring(lua_state.get(), -1);
38  LOGGER->critical("Failed to parse configuration file: {}", error);
39  throw lua::invalid_configuration_file(error);
40  }
41 
42  // FIXME: Find a better way of doing that
43 
44  // Read global configuration from global variable named 'configuration'
45  m_global_configuration.reset(new LazyConfigurationSet("configuration"));
46  int type = lua_getglobal(lua_state.get(), "configuration");
47  if (type == LUA_TTABLE) {
48  LOG(debug) << "Parsing global configuration.";
49  m_global_configuration->parse(lua_state.get(), -1);
50  }
51  lua_pop(lua_state.get(), 1);
52 
53  // Read vegas configuration
54  m_vegas_configuration.reset(new LazyConfigurationSet("vegas"));
55  type = lua_getglobal(lua_state.get(), "vegas");
56  if (type == LUA_TTABLE) {
57  LOG(debug) << "Parsing vegas configuration.";
58  m_vegas_configuration->parse(lua_state.get(), -1);
59  }
60  lua_pop(lua_state.get(), 1);
61 
62  for (auto& m: m_modules) {
63  LOG(debug) << "Configuration declared module " << m.type << "::" << m.name;
64 
65  lua_getglobal(lua_state.get(), m.type.c_str());
66  lua_getfield(lua_state.get(), -1, m.name.c_str());
67 
68  m.parameters = ConfigurationSet(m.type, m.name);
69  m.parameters.parse(lua_state.get(), -1);
70 
71  lua_pop(lua_state.get(), 2);
72  }
73 }
